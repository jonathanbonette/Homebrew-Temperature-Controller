294 void startNextRecipeStep(sc_integer recipeIndex) override
295 {
296 // Se a receita mudou ou é o início do processo (vinda do START_PROCESS)
297 if (this->currentRecipeIdx != recipeIndex)
298 {
299 this->currentRecipeIdx = recipeIndex;
300 this->currentStepIdx = 0; // Começa da primeira etapa
301 }
302 else
303 {
304 // Se já está na mesma receita, avança para a próxima etapa
305 this->currentStepIdx++;
306 }
307
308 if (this->currentRecipeIdx >= 0 && this->currentRecipeIdx < NUM_RECIPES)
309 {
310 const Recipe &recipe = recipes[this->currentRecipeIdx];
311 if (this->currentStepIdx < recipe.numSteps)
312 {
313 const RecipeStep &step = recipe.steps[this->currentStepIdx];
314 Serial.printf("Callback: INICIANDO ETAPA %d/%d: %s (Temp: %dC, Tempo: %dmin)\n",
315 this->currentStepIdx + 1, recipe.numSteps, step.name.c_str(), step.temperature, step.duration);
316
317 ControlCommand controlCmd = {CMD_START_RECIPE_STEP};
318 controlCmd.recipeIndex = this->currentRecipeIdx;
319 controlCmd.stepIndex = this->currentStepIdx;
320 controlCmd.targetTemperature = step.temperature; // Passa a temperatura alvo
321 controlCmd.durationMinutes = step.duration; // Passa a duração
322 xQueueSend(xControlQueue, &controlCmd, portMAX_DELAY); // Envia o comando para a ControlTask
323
324 // Exibe o status inicial da etapa no display
325 // Os valores de temperatura atual e tempo restante serão atualizados por uma tarefa de controle
326 showProcessStatus(0, step.temperature, step.duration, 0, const_cast<sc_string>(step.name.c_str()), this->currentStepIdx + 1, recipe.numSteps, true);
327 }
328 else
329 {
330 // Este bloco não deveria ser alcançado se as guardas do Yakindu estiverem corretas -- Debug
331 // Significa que startNextRecipeStep foi chamado mas não há mais etapas
332 Serial.println("Callback: Erro lógico: startNextRecipeStep chamada sem mais etapas.");
333 }
334 }
335 else
336 {
337 Serial.println("Callback: Erro: Receita inválida em startNextRecipeStep.");
338 // TODO: Lógica de erro ou transição de falha na máquina de estados
339 }
340 }

616 void heat(sc_integer target_temp) override
617 { // Implementação da operação 'heat' do Yakindu
618 if (myStatechart != nullptr)
619 {
620 int minTemp = 25; // Temperatura mínima que o PWM representa 0%
621 int maxTemp = 100; // Temperatura máxima que o PWM representa 100%
622 int maxDutyCycle = (1 << myStatechart->getPwm_resolution_bits()) - 1; // Ex: 1023 para 10 bits
623
624 int dutyCycle_calculated = 0;
625 if (target_temp <= minTemp)
626 {
627 dutyCycle_calculated = 0;
628 }
629 else if (target_temp >= maxTemp)
630 {
631 dutyCycle_calculated = maxDutyCycle;
632 }
633 else
634 {
635 // Mapeamento linear: duty = (target - min) / (max - min) * maxDuty
636 dutyCycle_calculated = map(target_temp, minTemp, maxTemp, 0, maxDutyCycle);
637 }
638
639 controlHeaterPWM(dutyCycle_calculated); // Chamar a função de controle de PWM diretamente
640 Serial.printf("Callback: Operacao 'heat' chamada. Alvo Temp: %dC -> PWM Duty Calculado: %d (max %d)\n", target_temp, dutyCycle_calculated, maxDutyCycle);
641 }
642 else
643 {
644 Serial.println("Callback: ERRO: myStatechart é nullptr em heat().");
645 }
646 }
