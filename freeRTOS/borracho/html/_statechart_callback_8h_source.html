287 void startNextRecipeStep(sc_integer recipeIndex) override
288 {
289 // Se a receita mudou ou é o início do processo (vinda do START_PROCESS)
290 if (this->currentRecipeIdx != recipeIndex)
291 {
292 this->currentRecipeIdx = recipeIndex;
293 this->currentStepIdx = 0; // Começa da primeira etapa
294 }
295 else
296 {
297 // Se já está na mesma receita, avança para a próxima etapa
298 this->currentStepIdx++;
299 }
300
301 if (this->currentRecipeIdx >= 0 && this->currentRecipeIdx < NUM_RECIPES)
302 {
303 const Recipe &recipe = recipes[this->currentRecipeIdx];
304 if (this->currentStepIdx < recipe.numSteps)
305 {
306 const RecipeStep &step = recipe.steps[this->currentStepIdx];
307 Serial.printf("Callback: INICIANDO ETAPA %d/%d: %s (Temp: %dC, Tempo: %dmin)\n",
308 this->currentStepIdx + 1, recipe.numSteps, step.name.c_str(), step.temperature, step.duration);
309
310 ControlCommand controlCmd = {CMD_START_RECIPE_STEP};
311 controlCmd.recipeIndex = this->currentRecipeIdx;
312 controlCmd.stepIndex = this->currentStepIdx;
313 controlCmd.targetTemperature = step.temperature; // Passa a temperatura alvo
314 controlCmd.durationMinutes = step.duration; // Passa a duração
315 xQueueSend(xControlQueue, &controlCmd, portMAX_DELAY); // Envia o comando para a ControlTask
316
317 // Exibe o status inicial da etapa no display
318 // Os valores de temperatura atual e tempo restante serão atualizados por uma tarefa de controle
319 showProcessStatus(0, step.temperature, step.duration, 0, const_cast<sc_string>(step.name.c_str()), this->currentStepIdx + 1, recipe.numSteps, true);
320 }
321 else
322 {
323 // Este bloco não deveria ser alcançado se as guardas do Itemis estiverem corretas
324 // Significa que startNextRecipeStep foi chamado mas não há mais etapas
325 Serial.println("Callback: Erro lógico: startNextRecipeStep chamada sem mais etapas.");
326 }
327 }
328 else
329 {
330 Serial.println("Callback: Erro: Receita inválida em startNextRecipeStep.");
331 }
332 }

608 void heat(sc_integer target_temp) override
609 { // Implementação da operação 'heat' do Itemis
610 if (myStatechart != nullptr)
611 {
612 int minTemp = 25; // Temperatura mínima que o PWM representa 0%
613 int maxTemp = 100; // Temperatura máxima que o PWM representa 100%
614 int maxDutyCycle = (1 << myStatechart->getPwm_resolution_bits()) - 1; // Ex: 1023 para 10 bits
615
616 int dutyCycle_calculated = 0;
617 if (target_temp <= minTemp)
618 {
619 dutyCycle_calculated = 0;
620 }
621 else if (target_temp >= maxTemp)
622 {
623 dutyCycle_calculated = maxDutyCycle;
624 }
625 else
626 {
627 // Mapeamento linear: duty = (target - min) / (max - min) * maxDuty
628 dutyCycle_calculated = map(target_temp, minTemp, maxTemp, 0, maxDutyCycle);
629 }
630
631 controlHeaterPWM(dutyCycle_calculated); // Chamar a função de controle de PWM diretamente
632 Serial.printf("Callback: Operacao 'heat' chamada. Alvo Temp: %dC -> PWM Duty Calculado: %d (max %d)\n", target_temp, dutyCycle_calculated, maxDutyCycle);
633 }
634 else
635 {
636 Serial.println("Callback: ERRO: myStatechart é nullptr em heat().");
637 }
638 }
